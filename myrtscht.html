103 # the subroutine invoked if the option -p is defined
104 sub process_group_files {
105     my $etcpath    = shift;
106     my $options    = shift;
107     my $groupfiles = shift;
108     my $cwd        = getcwd;
109
110     # We have to compute some things therewith the printout looks
111     # nice later.
112     # The number of lines per sheet of A4 paper must be 67 and 67
113     # must be the number of lines per sheet of A4 paper as thou have
114     # to use A4 paper with a number of 13 chars per column to get
115     # nice results.
116     my $maxsetnr        = ($options->{r} || 2) * 2 - 1;
117     my $gamesheetheigth = $maxsetnr + 6;
118     my $gamesheets      = sprintf("%d", 67 / $gamesheetheigth);
119     my $newlines        = 67 - $gamesheets * $gamesheetheigth;
120     DEBUG "There will be $gamesheets per page, so we have to insert "
121         . "$newlines newlines after them. They are $gamesheetheigth "
122         . "lines high.";
123
124     # the format for the printout; Alas, I can't declare the
125     # variables in a smaller scope.
126     my($groupname, $member1, $id, $table, $member2, $roundcnt);
127     my $printoutform =
128           "format GAMEPRINT = \n"
129         . 'Group: @' . '<' x 36 . "\n"
130         . '$groupname' . "\n"
131         . 'Round: @<<<< Table: @<<<<' . ' ' x 10 . 'ID: @<<<<' . "\n"
132         . '$roundcnt, $table, $id' . "\n"
133         . '-' x 22 . '+' . '-' x 21 . "\n"
134         . '@' . '|' x 20 . ' |' . '@' . '|' x 20 . "\n"
135         . '$member1, $member2' . "\n"
136         . '-'  x 22 . '+' . '-' x 21 . "\n"
137         . (' ' x 22 . '|' . ' ' x 21 . "\n") x $maxsetnr
138         . '-'  x 22 . '+' . '-' x 21 . "\n"
139         . ".\n";
140     DEBUG $printoutform if defined $options->{d};
141     eval $printoutform;
142     die $@ if $@;
143
144     # process the group files one by one
145     foreach my $groupfile (@$groupfiles) {
146         my $printfile = "gamesheets";
147         my $groupfnr = substr($groupfile, 0, 2);
148         my $printer;
149
150         # manage the group directory
151         {
152         my $groupdir = $groupfnr . ".d";
153         mkdir($groupdir, 0755)
154             or die "Couldn't make directory $groupdir: $!\n";
155         rename($groupfile, File::Spec->catfile($groupdir, $groupfile))
156             or die "Couldn't move $groupfile to $groupdir: $!\n";
157         chdir $groupdir
158             or die "Couldn't change into $groupdir: $!";
159         DEBUG "Now in group directory $groupdir.";
160         }
161
162         # read the groupfile
163         open(GROUPFILE, "<", $groupfile)
164             or die "Couldn't open group file $groupfile: $!\n";
165         chomp(my @gf_content = <GROUPFILE>);
166         close GROUPFILE;
167         DEBUG "Group file content: @gf_content";
168         $groupname  = substr(shift @gf_content, 6);
169         DEBUG "The group is named $groupname.";
170         my @tables  = split(m([ ,;.:/]+), substr(shift @gf_content, 8));
171         DEBUG "It plays at the tables @tables.";
172         shift @gf_content;
173         my @members = @gf_content;
174         DEBUG "It consists of @members.";
175         undef @gf_content;
176
177         # initialize the printing things
178         # invoke the spooler or not
179         unless (defined $options->{s}) {
180             my $cups = Net::CUPS->new()
181                 or warn "Couldn't get a nice CUPS - we "
182                       . "won't be able to print.\n";
183             $printer = $cups->getDestination($options->{P})
184                 or warn "Couldn't get your printer's destination: "
185                       . "You have to specify another.\n";
186             $printer->addOption("cpi", 13);
187             $printer->addOption("columns", 2);
188         }
189         open(GAMEPRINT, ">>", $printfile)
190             or die "Couldn't open $printfile: $!\n";
191         select((select(GAMEPRINT), $|++)[0]);
192
193         # create the schedule for the group
194         my @schedule;
195         {
196         DEBUG "The schedule:";
197         my $group = Games::Tournament::RoundRobin->new(
198                 league => \@members
199                 );
200         for (my $i = 1; $i <= $group->rounds(); $i++) {
201             my %round;
202             my %raw_schedule = %{ $group->membersInRound($i) };
203             DEBUG "Round $i.";
204             my %parsed_values;
205             foreach my $member (keys %raw_schedule) {
206                 next if $member eq "Bye";
207                 next if $raw_schedule{$member} eq "Bye";
208                 next if $parsed_values{$member};
209                 $round{$member} = $raw_schedule{$member};
210                 $parsed_values{$round{$member}} = 1;
211             }
212             push(@schedule, \%round);
213             DEBUG "@schedule";
214         }
215         }
216
217         # process the schedule and create the necessary files
218         {
219         my($gamecnt, $appendcnt);
220
221         for my $round (@schedule) {
222             $roundcnt++;
223             for my $member (keys %{ $round }) {
224                 $id      = $groupfnr . ++$gamecnt;
225                 $table   = shift @tables;
226                 $member1 = $member;
227                 $member2 = $round->{$member};
228                 my $gamefile = sprintf("%02d.s", $gamecnt);
229                 push(@tables, $table);
230
231                 # create the gamefile
232                 open(GAMEFILE, ">", $gamefile)
233                     or die "Couldn't open $gamefile: $!\n";

235
236                 # append to the print file
237                 write GAMEPRINT;
238                 $appendcnt++;
239                 if ($appendcnt == $gamesheets) {
240                     print GAMEPRINT "\n" x $newlines;
241                     $appendcnt = 0;
242                 }
243             }
244         }
245         }
246
247         # print out or do something else
248         unless (defined $options->{s}) {
249             $printer->printFile($printfile, $printfile)
250                 or die "Couldn't print out: $!\n";
251             unlink $printfile
252                 or warn "Couldn't remove $printfile: $!\n";
253         }
254
255         chdir $cwd
256             or die "Couldn't change into $cwd: $!\n";
257         DEBUG "Back in $cwd.";
258     }
259     INFO "Created the schedule.";
260 }

268     my $file_containing = shift;

270     # process the input and filtering out just the game files to

289 # process the game files
290 sub process_game_files {
291     my $etcpath   = shift;
292     my $options   = shift;
293     my $gamefiles = shift;
294     my %players;
295
296     # process the game file and create the data structure
297     for my $gamefile (@$gamefiles) {
298         open(GAMEFILE, $gamefile)
299             or die "Couldn't open game file $gamefile: $!\n";
300         my($name1, $name2);
301         my $numcont;
302
303         while (<GAMEFILE>) {
304             chomp;
305             next if /#/;
306             if (/(.+) \| (.+)/) {
307                 ($name1, $name2) = ($1, $2);
308                  DEBUG "The players are $name1 and $name2.";
309                  $numcont = 0;
310             } else {
311                 ($players{$name1}{balls}, $players{$name2}{balls})
312                     = /(\d+)\D+(\d+)/;
313                 DEBUG "Result: $1 : $2";
314                 $numcont = 1;
315             }
316
317             $players{$name1}{all_fails} += $players{$name2}{balls},
318             $players{$name2}{all_fails} += $players{$name1}{balls},
319             $players{$name1}{all_balls} += $players{$name1}{balls},
320             $players{$name2}{all_balls} += $players{$name2}{balls},
321             $players{$name1}{ball_difference} +=
322               ($players{$name1}{balls} - $players{$name2}{balls}),
323             $players{$name2}{ball_difference} +=
324               ($players{$name2}{balls} - $players{$name1}{balls}),
325                 if $numcont;
326
327             if ($numcont and
328                     $players{$name1}{balls} > $players{$name2}{balls}) {
329                 $players{$name1}{sets_won}++;
330                 $players{$name1}{all_sets_won}++;
331                 $players{$name2}{sets_lost}++;
332                 $players{$name2}{all_sets_lost}++;
333             } elsif ($numcont) {
334                 $players{$name2}{sets_won}++;
335                 $players{$name2}{all_sets_won}++;
336                 $players{$name1}{sets_lost}++;
337                 $players{$name1}{all_sets_lost}++;
338             }
339             $players{$name1}{balls} = 0; # Just for this game
340             $players{$name2}{balls} = 0; #
341         }
342
343             if ($players{$name1}{sets_won}
344                     > $players{$name2}{sets_won}) {
345                 $players{$name1}{wins}++;
346                 $players{$name2}{losts}++;
347             } else {
348                 $players{$name2}{wins}++;
349                 $players{$name1}{losts}++;
350             }
351
352             $players{$name1}{set_difference} +=
353               ($players{$name1}{sets_won} - $players{$name2}{sets_won});
354             $players{$name2}{set_difference} +=
355               ($players{$name2}{sets_won} - $players{$name1}{sets_won});
356
357             $players{$name1}{sets_won} = 0; # Just for this game.
358             $players{$name2}{sets_won} = 0; #
359     }
360     # I know, this whole thing is very very ugly. But how ugly
361     # would it be  if there were no hashes? On top of that it
362     # produces occasional warnings. But I think, if it
363     # wouldn't produce occasional warnings, it would also be
364     # far more ugly.
365     close GAMEFILE;
366     INFO "Processed the game files.";
367
368     # sort the people according to their games
369     my @ranking = sort {
370         $players{$b}{wins}           <=> $players{$a}{wins}           or
371         $players{$a}{losts}          <=> $players{$b}{losts}          or
372         $players{$b}{set_difference} <=> $players{$a}{set_difference} or
373         $players{$b}{ball_difference} <=> $players{$a}{ball_difference}
374     } keys %players;
375     DEBUG "Sorted all the stuff.";
376     
377     # generate the ranking file
378     open(RANKING, ">", "ranking")
379         or die "Couldn't open \"ranking\" for writing: $!\n";
380     printf RANKING ("%4s %-21s\t%-11s\t%-11s\t%-5s\t%-11s\t%-5s\n",
381             "Rank", "Name", "Games", "Sets", "SD", "Balls", "BD");
382     my $rank;
383     foreach my $player (@ranking) {
384         printf RANKING (
385                     "%3d. %-21s\t%4d : %4d\t%4s : %4s\t%5s\t"
386                   . "%4s : %4s\t%5s\n",
387                     ++$rank, $player,
388                     $players{$player}{wins}            || 0,
389                     $players{$player}{losts}           || 0,
390                     $players{$player}{all_sets_won}    || 0,
391                     $players{$player}{all_sets_lost}   || 0,
392                     $players{$player}{set_difference}  || 0,
393                     $players{$player}{all_balls}       || 0,
394                     $players{$player}{all_fails}       || 0,
395                     $players{$player}{ball_difference} || 0,
396                     );
397     }
398     INFO "Generated the ranking file.";
399 }

405 =head1 NAME
406
407 myrtscht - My Round robin Tournament SCHeduling for Table tennis
408
409
410 =head1 SYNOPSIS
411
412 myrtscht B<-n> [I<name>] [B<-g> I<number>] [B<-d> [I<file>]]
413
414 myrtscht B<-p> I<files> [B<-r> I<rounds>] [B<-s>] [B<-P> I<printer>] [B<-d> [I<file>]]
415
416 myrtscht B<-a> I<files>/I<directory> [B<-d> [I<file>]]
417
418 myrtscht B<--help>
419
420 myrtscht B<-h>
421
422
423
424 =head1 DESCRIPTION
425
426 B<myrtscht> is a program which helps you managing table tennis
427 tournaments with the round robin system.  But I think, you can also
428 use it for other games.  It is completely text and command based, so
429 there is neither a CLI nor a GUI, my apologies for this (if this needs
430 to be apologised).  Therefore it works, as you can see in the
431 synopsis, with running commands and editing text files.
432
433
434 =head1 OPTIONS AND ARGUMENTS
435
436 =over
437
438 =item B<-n> [I<name>]
439
440 Creates a new directory used for the tournament with the group files in
441 it.  You can also give the directory a name, which has no effects on the
442 group files, the default is "Tournament-<date>".
443
444 =item B<-g> I<number>
445
446 Defines the number of group files created when running with B<-n>, the
447 default is 10.
448
449 =item B<-p> I<files>
450
451 Creates a group directory with the game files in it from each of the
452 I<files> given and prints out some nice formatted tables on sheets of
453 paper where you can write down the results of the games.
454
455 =item B<-r> I<sets>
456
457 Defines the number of sets to win for winning the whole game, the
458 default is 2.  This is required when running with B<-p>.
459
460 =item B<-s>
461
462 Disables printing when running with B<-p>.
463
464 =item B<-P> I<printer>
465
466 Defines the I<printer> used when running with B<-p>, the default is the
467 standard printer.
468
469 =item B<-a> I<files>/I<directory>
470
471 Computes a ranking file from the given group I<files> or I<directory>.
472
473 =item B<-d> [I<file>]
474
475 Switches to debugging mode.  The log is created by using the
476 L<Log::Log4perl(3)|Log::Log4perl> module.  You can specify a config
477 I<file> to modify the output.
478
479 =item B<-h> or B<--help>
480
481 Print this help.
482
483 =back

486 =head1 FILES
487
488 B<myrtscht> creates different files in its process of working.  They all
489 have a special extension.  These may show up:
490
491 =head2 Group Files F<*.g>
492
493 These are the files you are going to see first when running a
494 tournament.  Their format (before editing) is:
495
496     Name:
497     Tables:
498     Members:
499 C< >
500
501 Behind "Name: " you have to write the name of the group.  Behind
502 "Tables: " the names of the tables (numbers are recommended) the group
503 plays at stand in a comma separated list. (Yes there are also other
504 separators possible.  Look in the source code if you want them
505 implicitly!)  I<Under> "Members:" you have to write the names of the
506 members of the group.
507
508 Example:
509
510     Name: 50 before Christ
511     Tables: 1, 2, 3
512     Members:
513     Asterix
514     Obelix
515     Gaius Iulius Caesar
516     Gaius Bonus
517
518 Note: Names with more than 20 letters are unbearable. (Maybe also the
519 parents of the persons with this names.)
520
521 =head2 Game Files F<*.s>
522
523 These are the files being in a group directory after running with B<-p>.
524 They look like this (before editing):
525
526     Asterix | Gaius Iulius Caesar
527
528 There you can write down the results of a match.
529
530 Example:
531
532     Asterix | Gaius Iulius Caesar
533     11  8
534     10  12
535     11  9
536
537 Note: It is recommend to use just whitespace between the numbers.
